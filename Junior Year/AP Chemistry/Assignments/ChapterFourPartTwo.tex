  \item Write a net ionic equation for any precipitation reaction that occurs when 1[\si{\Molar}] solutions of the following are mixed:

    \begin{enumerate}

      \item \ce{Na3PO4} and \ce{BaCl2} \eqref{1}

        \begin{equation}
          \begin{split}
            \ce{3Ba^2+(aq) + 2PO4^3-(aq) -> Ba3(PO4)2(s)}
          \end{split}
          \label{1}
        \end{equation}

      \item \ce{ZnSO4} and \ce{KOH} \eqref{2}

        \begin{equation}
          \begin{split}
            \ce{Zn^2+(aq) + 2OH-(aq) -> 2ZnOH(s)}
          \end{split}
          \label{2}
        \end{equation}

      \item \ce{(NH4)2SO4} and \ce{NaCl} \eqref{3}

        \begin{equation}
          \begin{split}
            \text{No reaction because neither Ammonium nor Sulfate are soluble}
          \end{split}
          \label{3}
        \end{equation}

      \item \ce{Co(NO3)3} and \ce{Na3PO4} \eqref{4}

        \begin{equation}
          \ce{Co^3+(aq) + SO4^3-(aq) -> CoPO4(s)}
          \label{4}
        \end{equation}

    \end{enumerate}

\end{enumerate}
