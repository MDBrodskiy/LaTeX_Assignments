  \item Assign oxidation numbers to each element in:

    \begin{enumerate}

      \item \ce{NO} $\rightarrow$ $\ce{N} = 2,\,\ce{O}=-2$

      \item \ce{NH3} $\rightarrow$ $\ce{N} = -3,\,\ce{H}=1$

      \item \ce{K2O2} $\rightarrow$ $\ce{K} = 1,\,\ce{O}=-1$

      \item \ce{ClO3-} $\rightarrow$ $\ce{Cl} = 5,\,\ce{O}=-2$

    \end{enumerate}

  \item Assign oxidation numbers to each element in:

  \begin{enumerate}

      \item \ce{CH4} $\rightarrow$ $\ce{H} = 1,\,\ce{C}=-4$

      \item \ce{CO3^2-} $\rightarrow$ $\ce{C} = 4,\,\ce{O}&=-2$

      \item \ce{IO4-} $\rightarrow$ $\ce{I} = 7,\,\ce{O}=-2$

      \item \ce{N2H4} $\rightarrow$ $\ce{H} = 1,\,\ce{N}=-2$

    \end{enumerate}

  \item Assign oxidation numbers to each element in:

  \begin{enumerate}

      \item \ce{ClO3-} $\rightarrow$ $\ce{Cl} = 5,\,\ce{O}=-2$

      \item \ce{H2SO3} $\rightarrow$ $\ce{H} = 1,\,\ce{O}=-2,\,\ce{S}=4$

      \item \ce{K2O2} $\rightarrow$ $\ce{K} = 1,\,\ce{O}=-1$

      \item \ce{Na3N} $\rightarrow$ $\ce{Na} = 1,\,\ce{N}=-3$

    \end{enumerate}

  \item Assign oxidation numbers to each element in:

  \begin{enumerate}

      \item \ce{HIO3} $\rightarrow$ $\ce{H} = 1,\,\ce{O}=-2,\,\ce{I}=5$

      \item \ce{NaMnO4} $\rightarrow$ $\ce{Na} = 1,\,\ce{O}=-2,\,\ce{Mn}=7$

      \item \ce{SnO2} $\rightarrow$ $\ce{Sn} = 4,\,\ce{O}=-2$

      \item \ce{NOF} $\rightarrow$ $\ce{O} = -2,\,\ce{N}=3,\,\ce{F}=-1$

      \item \ce{NaO2} $\rightarrow$ $\ce{Na} = 1,\,\ce{O}=-\frac{1}{2}$

    \end{enumerate}


\end{enumerate}
