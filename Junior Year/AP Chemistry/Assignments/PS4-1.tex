\begin{enumerate}

  \item Write the molecular, complete ionic, and net equations for the following:

    \begin{enumerate}

      \item \ce{BaCl2} and \ce{Na2SO4} \eqref{1}

        \begin{equation}
          \begin{split}
            \ce{BaCl2(aq) + Na2SO4(aq) -> } & \ce{2NaCl(aq) + BaSO4(s)}\\
            \hline
            & \\
            \ce{Ba^2+(aq) + 2Cl-(aq) + } & \ce{2Na+(aq) + SO4^2-(aq) ->} \\
            & \ce{2Cl-(aq) + 2Na+(aq) + BaSO4(s)} \\
            \hline
            & \\
            \ce{Ba^2+(aq) + SO4^2-(aq) -> } & \ce{BaSO4(s)} \\
            \hline
            & \\
          \end{split}
          \label{1}
        \end{equation}

      \item \ce{KOH} and \ce{Fe(NO3)3} \eqref{2}

        \begin{equation}
          \begin{split}
            \ce{3KOH(aq) + Fe(NO3)3(aq) -> } & \ce{3KNO3(aq) + Fe(OH)3(s)}\\
            \hline
            & \\
            \ce{3K+(aq) + 3OH-(aq) + } & \ce{3NO3-(aq) + Fe^3+(aq) ->}\\
            & \ce{3K+(aq) + 3NO3-(aq) + Fe(OH)3(s)}\\
            \hline
            & \\
            \ce{3OH-(aq) + Fe^3+(aq) -> } & \ce{Fe(OH)3(s)}\\
            \hline
            & \\
          \end{split}
          \label{2}
        \end{equation}

      \item \ce{Ca(C2H3O2)2} and \ce{K2SO4} \eqref{3}

        \begin{equation}
          \begin{split}
            \ce{Ca(C2H3O2)2(aq) + K2SO4(aq)} & \ce{ -> 2KC2H3O2(aq) + CaSO4(s)}\\
            \hline
            & \\
            \ce{Ca^2+(aq) + 2C2H3O2-(aq)} & \ce{ + 2K+(aq) + SO4^2-(aq) ->} \\
            & \ce{2C2H3O2-(aq) + 2K+(aq) + CaSO4(s)}\\
            \hline
            & \\
            \ce{Ca^2+(aq) + SO4^2-(aq) -> } & \ce{CaSO4(s)}\\
            \hline
            & \\
          \end{split}
          \label{3}
        \end{equation}


      \item \ce{FeCl3} and \ce{NaOH} \eqref{4}

        \begin{equation}
          \begin{split}
            \ce{3NaOH(aq) + FeCl3(aq) ->} & \ce{Fe(OH)3(s) + 3NaCl(aq)} \\
            \hline
            & \\
            \ce{Fe^3+(aq) + 3Cl-(aq) + } & \ce{3Na+(aq) + 3OH-(aq) -> }\\
            & \ce{3Cl-(aq) + 3Na+(aq) + Fe(OH)3(s)}\\
            \hline
            & \\
            \ce{Fe^3+(aq) + 3OH-(aq) ->} & \ce{Fe(OH)3(s)}\\
            \hline
            & \\
          \end{split}
          \label{4}
        \end{equation}

    \end{enumerate}

  \item What volume of $.25[\si{\Molar}]$ \ce{AgNO3} is required to react with:

    \begin{enumerate}

      \item $82[\si{\milli\liter}]$ of $.08[\si{\Molar}]$ \ce{Na2S} \eqref{5}

        \begin{equation}
          \begin{split}
            82[\si{\milli\liter}] & = .082[\si{\liter}]\\
            \ce{2AgNO3(aq) + Na2S(aq)} & \ce{ -> AgS(s) + 2NaNO3(aq)}\\
            .082\cdot.08 = .00656[\si{\mole}_{\ce{Na2S}}] & \rightarrow .01312[\si{\mole}_{\ce{AgNO3}}]\\
            \frac{.01312}{.25} & = .052[\si{\liter}_{\ce{AgNO3}}]\\
          \end{split}
          \label{5}
        \end{equation}
        
      \item $15[\si{\milli\liter}]$ of $.44[\si{\Molar}]$ \ce{K2CrO4} \eqref{6}

        \begin{equation}
          \begin{split}
            15[\si{\milli\liter}] & = .015[\si{\liter}]\\
            \ce{2AgNO3(aq) + K2CrO4(aq)} & \ce{ -> Ag2CrO4(s) + 2KNO3(aq)}\\
            .015\cdot.44 = .0066[\si{\mole}_{\ce{K2CrO4}}] & \rightarrow .0132[\si{\mole}_{\ce{AgNO3}}]\\
            \frac{.0132}{.25} & = .053[\si{\liter}_{\ce{AgNO3}}]\\
          \end{split}
          \label{6}
        \end{equation}

    \end{enumerate}
