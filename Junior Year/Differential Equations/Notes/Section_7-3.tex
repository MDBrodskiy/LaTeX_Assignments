\begin{itemize}

  \item If $\mathcal{L}\{f(t)\}=F(s)$ and $a$ is any real number, then:

    \begin{equation}
      \mathcal{L}\{e^{at}f(t)\}=F(s-a)
      \label{1}
    \end{equation}

  \item This theorem works both ways, as:

    \begin{equation}
      e^{at}f(t)=\mathcal{L}^{-1}\{F(s-a)\}
      \label{2}
    \end{equation}

\end{itemize}
