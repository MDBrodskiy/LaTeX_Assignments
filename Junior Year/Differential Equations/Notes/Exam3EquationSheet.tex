  \begin{equation*}
    \begin{split}
      \text{Laplace Transform Definition:}\\
      \mathcal{L}\left\{ f(t) \right\}=\int_0^{\infty}e^{-st}f(t)\,dt
      \end{split}
    \label{1}
  \end{equation*}

  \begin{equation*}
    \begin{split}
      \text{Unit Step Definition:}\\
      \mathcal{U}(t-a)=\left\{\begin{array}{ll} 0, & 0\leq t < a\\ 1, & t \geq a \\ \end{array}
    \end{split}
    \label{2}
  \end{equation*}
