\begin{itemize}

  \item A first-order linear differential equation takes the form:

    $$a_1(x)\frac{dy}{dx}+a_0(x)y=g(x)$$

  \item The standard form (where $P(x)=\frac{a_0(x)}{a_1(x)}$ and $f(x)=g(x)$) is as follows:

      $$\frac{dy}{dx}+P(x)y=f(x)$$

\end{itemize}
