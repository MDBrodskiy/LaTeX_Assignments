\newcommandx{\grid}[4][]{
  %% draw grid
  \draw[color=cqcqcq,, xstep=1.0cm,ystep=1.0cm] (#1,#3) grid (#2,#4);
}


\newcommandx{\ylabels}[2][2=\xscalefactor]{
  \foreach \y in {#1}{
    \draw[shift={(0,\y)},color=black]
                (2pt/#2,0pt) -- (-2pt/#2,0pt) node[left] {\footnotesize $\y$};
  }

}



\newcommandx{\xlabels}[2][2=\yscalefactor]{
  %% draw labels on x-axis
  \foreach \x in {#1}{
    \draw[shift={(\x,0)},color=black]
      (0pt,2pt/#2) -- (0pt,-2pt/#2) node[below] {\footnotesize $\x$};
  }

  % label origin
  %\draw[color=black] (-1.5pt, -2pt/#2 + 2pt) node[below right] {\footnotesize $0$};
}



\newcommandx{\axes}[4]{
  %% draw x-axis
  \draw[<->,color=black] (#1,0) -- (#2,0);% node[below] {\footnotesize $x$};

  %% draw x label
  \draw (#2, -2pt- 0.2ex) node[below] {\footnotesize $x$};

  %% draw y-axis
  \draw[<->,color=black] (0,#3) -- (0,#4);% node[left] {\footnotesize $y$};

  %% draw y label
  \draw (-2pt, #4) node[left] {\footnotesize $y$};
}


\newcommandx{\xypoint}[4][1=color1, 4=3.0pt]{
  \fill[color=#1] (#2,#3) ellipse ({#4/\xscalefactor} and {#4/\yscalefactor})
}


\newcommandx{\openpoint}[4][1=color1, 4=4.0pt]{
   \fill[color=#1] (#2,#3) circle (#4);
   \fill[color=white] (#2,#3) circle (#4 - 0.8pt)
 }


 \newcommandx{\linesegment}[6][1=-, 6=color1]{
   \draw[#1, color=#6, thick] (#2, #3) -- (#4, #5);
}

\newcommandx{\origin}[4][1=-1pt, 2=-1.0pt, 3=\xscalefactor, 4=\yscalefactor]{
  %% label origin
  \draw[color=black] ( #1 * \xscalefactor , #2 * \yscalefactor ) node[below right] {\footnotesize $0$};

}
